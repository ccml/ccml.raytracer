   Given g ← group()
     And s1 ← sphere()
     And s2 ← sphere()
     And set_transform(s2, translation(0, 0, -3))
     And s3 ← sphere()
     And set_transform(s3, translation(5, 0, 0))
     And add_child(g, s1)
     And add_child(g, s2)
     And add_child(g, s3)
    When r ← ray(point(0, 0, -5), vector(0, 0, 1))
     And xs ← local_intersect(g, r)
    Then xs.count = 4
     And xs[0].object = s2
     And xs[1].object = s2
     And xs[2].object = s1
     And xs[3].object = s1
